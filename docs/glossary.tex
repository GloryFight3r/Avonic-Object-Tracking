\newglossaryentry{active-speaker}
{
    name=active speaker,
    description={A person who is 
currently speaking and the Sennheiser microphone is able to detect their voice and their voice volume is 
greater than the required threshold we have set}
}


\newglossaryentry{speaker-plane}
{
    name=speaker plane,
    description={An imaginary 2D plane in the room, on which speakers' mouths are located (roughly, depending on height of the speaker)}
}

\newglossaryentry{microphone-height}
{
    name=microphone height,
    description={The distance between the speaker plane and the plane on which the microphone is located}
}

\newglossaryentry{calibration}
{
    name=calibration,
    description={During the calibration stage, we calculate the distance from the camera to
      the microphone and obtain the camera's relative coordinates to the microphone}
}

\newglossaryentry{microphone}
{
  name=microphone,
  description={Technically, the Sennheiser TeamConnect Ceiling 2 is a microphone array,
  but since the individual microphones inside it are inaccessible,
we treat the whole array as one microphone, and we refer to it as one.
Unless specified otherwise, the word microphone will refer to the entire Ceiling 2}
}

\newglossaryentry{integration}
{
    name=integration,
    description={We reffer to the GeneralController object as integration}
}
